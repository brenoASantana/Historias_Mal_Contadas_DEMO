#include <stdio.h>

void limparTerminal() {
    printf("\033[H\033[J"); // Sequência ANSI para limpar o terminal
}

int main() {
    // Antes de limpar
    printf("Este é o conteúdo antes de limpar o terminal.\n");
    
    // Limpar o terminal
    limparTerminal();
    
    // Depois de limpar
    printf("Este é o conteúdo depois de limpar o terminal.\n");
    
    return 0;
}
